---------------------------------------------
.. code-block:: c++

    #include <type_traits>
    using namespace std;

    int main() {
        vector<int> arr(5);
        // 注意使用尖括号
        cout << is_same_v<decltype(arr.size()), unsigned long> << endl;
    }
